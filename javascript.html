JavaScript คืออะไร?

JavaScript เป็นภาษาการเขียนโปรแกรมที่ใช้ในการพัฒนาเว็บไซต์ โดยเฉพาะสำหรับการจัดการกับองค์ประกอบต่าง ๆ บนหน้าเว็บและเพิ่มฟังก์ชันต่าง ๆ ที่ไม่สามารถทำได้ด้วย HTML และ CSS เพียงอย่างเดียว โดย JavaScript ทำงานร่วมกับ HTML และ CSS เพื่อสร้างประสบการณ์ที่มีความอินเทอร์แอคทีฟและตอบสนองต่อการกระทำของผู้ใช้ได้อย่างมีประสิทธิภาพ

คำสั่งพื้นฐานใน JavaScript

1. การใช้คำสั่ง console.log()

คำสั่ง console.log() ใช้สำหรับพิมพ์ข้อความไปยังคอนโซล ซึ่งเป็นเครื่องมือที่ช่วยในการดีบักโปรแกรม:

console.log("Hello, World!");

ในตัวอย่างนี้ ข้อความ "Hello, World!" จะถูกพิมพ์ลงในคอนโซล

2. การประกาศตัวแปร

สามารถใช้ var, let, หรือ const เพื่อประกาศตัวแปร:

let message = "Hello, JavaScript!";
const pi = 3.14;
var age = 25;

- let ใช้สำหรับประกาศตัวแปรที่สามารถเปลี่ยนค่าได้
- const ใช้สำหรับประกาศตัวแปรที่ไม่สามารถเปลี่ยนค่าได้
- var ใช้สำหรับประกาศตัวแปรที่สามารถเปลี่ยนค่าได้ แต่ไม่แนะนำใน ES6 ขึ้นไป

3. การใช้คำสั่ง if และ else

คำสั่ง if และ else ใช้ในการตัดสินใจตามเงื่อนไข:

let number = 10;

if (number > 0) {
    console.log("Number is positive.");
} else if (number < 0) {
    console.log("Number is negative.");
} else {
    console.log("Number is zero.");
}

โปรแกรมนี้จะตรวจสอบค่าของตัวแปร number และพิมพ์ข้อความที่เหมาะสมตามเงื่อนไขที่กำหนด

4. การใช้คำสั่ง for

คำสั่ง for ใช้สำหรับการวนลูปในจำนวนรอบที่กำหนด:

for (let i = 1; i <= 5; i++) {
    console.log("Iteration " + i);
}

โปรแกรมนี้จะพิมพ์ข้อความ "Iteration" ตามด้วยหมายเลขลูปตั้งแต่ 1 ถึง 5

5. การใช้คำสั่ง while

คำสั่ง while ใช้สำหรับการวนลูปตราบใดที่เงื่อนไขยังเป็นจริง:

let i = 1;
while (i <= 5) {
    console.log("Iteration " + i);
    i++;
}

โปรแกรมนี้จะทำการพิมพ์ข้อความ "Iteration" ตามด้วยหมายเลขลูปตั้งแต่ 1 ถึง 5 เช่นเดียวกับคำสั่ง for

6. การใช้คำสั่ง switch

คำสั่ง switch ใช้สำหรับการตรวจสอบค่าของตัวแปรและดำเนินการตามกรณีที่กำหนด:

let day = 3;

switch (day) {
    case 1:
        console.log("Monday");
        break;
    case 2:
        console.log("Tuesday");
        break;
    case 3:
        console.log("Wednesday");
        break;
    case 4:
        console.log("Thursday");
        break;
    case 5:
        console.log("Friday");
        break;
    case 6:
        console.log("Saturday");
        break;
    case 7:
        console.log("Sunday");
        break;
    default:
        console.log("Invalid day");
}

โปรแกรมนี้จะพิมพ์ชื่อของวันในสัปดาห์ตามค่าของตัวแปร day โดยใช้คำสั่ง switch

7. การประกาศฟังก์ชัน

ฟังก์ชันใน JavaScript ใช้สำหรับการรวบรวมกลุ่มของคำสั่งที่ต้องการทำงานร่วมกัน:

function greet(name) {
    return "Hello, " + name + "!";
}

console.log(greet("Alice"));

ฟังก์ชัน greet จะรับชื่อและส่งคืนข้อความทักทาย

8. การทำงานกับอาร์เรย์

อาร์เรย์เป็นรายการของค่า:

let fruits = ["Apple", "Banana", "Cherry"];

fruits.forEach(function(fruit) {
    console.log(fruit);
});

โปรแกรมนี้จะพิมพ์ชื่อผลไม้ทั้งหมดในอาร์เรย์ fruits

9. การจัดการกับอ็อบเจ็กต์

อ็อบเจ็กต์ใน JavaScript ใช้สำหรับเก็บข้อมูลที่มีลักษณะเป็นคู่ของชื่อและค่า:

let person = {
    firstName: "John",
    lastName: "Doe",
    age: 30,
    greet: function() {
        return "Hello, " + this.firstName + " " + this.lastName;
    }
};

console.log(person.greet());

ในตัวอย่างนี้ อ็อบเจ็กต์ person มีคุณสมบัติ firstName, lastName, และ age และมีเมธอด greet ที่ส่งคืนข้อความทักทาย
